```python
async def search_all_sources(query: str, top_k: int = 10):
    """РџРѕРёСЃРє РїРѕ РІСЃРµРј РёСЃС‚РѕС‡РЅРёРєР°Рј СЃ РѕР±СЉРµРґРёРЅРµРЅРёРµРј СЂРµР·СѓР»СЊС‚Р°С‚РѕРІ."""
    # РџР°СЂР°Р»Р»РµР»СЊРЅС‹Р№ Р·Р°РїСЂРѕСЃ РєРѕ РІСЃРµРј РёРЅРґРµРєСЃР°Рј
    tasks = [
        index.search(query_vector, k=top_k)
        for index in self.indexes.values()
    ]
    results = await asyncio.gather(*tasks)
    
    # РћР±СЉРµРґРёРЅСЏРµРј РІСЃРµ СЂРµР·СѓР»СЊС‚Р°С‚С‹
    all_results = []
    for source_results, source_name in zip(results, self.indexes.keys()):
        metadata = self.metadata[source_name]
        for vid, score in zip(*source_results):
            all_results.append(
                VectorSearchResult(
                    id=metadata[vid]["document_id"],
                    score=score,
                    content=metadata[vid]["content"],
                    metadata=metadata[vid],
                    source=source_name
                )
            )
    
    # РЎРѕСЂС‚РёСЂСѓРµРј РїРѕ score Рё РІРѕР·РІСЂР°С‰Р°РµРј С‚РѕРї-k
    all_results.sort(key=lambda x: x.score, reverse=True)
    return all_results[:top_k]
```

**Р’СЂРµРјСЏ:** ~150ms (3 Г— ~50ms РїР°СЂР°Р»Р»РµР»СЊРЅРѕ)

```python
async def search_with_filters(
    query: str,
    top_k: int = 10,
    filters: Optional[Dict] = None
):
    """РџРѕРёСЃРє СЃ С„РёР»СЊС‚СЂР°РјРё."""
    if not filters:
        return await self.search_all_sources(query, top_k)
    
    # Р•СЃР»Рё РµСЃС‚СЊ С„РёР»СЊС‚СЂ РїРѕ source вЂ” РёС‰РµРј С‚РѕР»СЊРєРѕ РІ РЅС‘Рј
    if "source" in filters:
        sources = filters["source"]
        results = []
        for source in sources:
            source_results = await self.search_by_source(query, source, top_k)
            results.extend(source_results)
        
        results.sort(key=lambda x: x.score, reverse=True)
        return results[:top_k]
    
    # Р”СЂСѓРіРёРµ С„РёР»СЊС‚СЂС‹ (category, tags, date) вЂ” РїРѕСЃС‚-С„РёР»СЊС‚СЂР°С†РёСЏ
    all_results = await self.search_all_sources(query, top_k * 3)
    filtered = [r for r in all_results if matches_filters(r, filters)]
    return filtered[:top_k]
```

```python
    """Р”РѕР±Р°РІР»РµРЅРёРµ РґРѕРєСѓРјРµРЅС‚Р° РІ СЃРѕРѕС‚РІРµС‚СЃС‚РІСѓСЋС‰РёР№ РёРЅРґРµРєСЃ."""
```

```python
    # РЎРѕР·РґР°С‘Рј СЂР°Р·РґРµР»СЊРЅС‹Рµ РёРЅРґРµРєСЃС‹
    for source, items in by_source.items():
        # РЎРѕР·РґР°С‘Рј РЅРѕРІС‹Р№ РёРЅРґРµРєСЃ
        new_index = faiss.IndexFlatIP(DIMENSION)
        new_metadata = {"vectors": {}, "documents": {}}
        
        for vector_id, meta in items:
            # РљРѕРїРёСЂСѓРµРј РІРµРєС‚РѕСЂ
            vector = index.reconstruct(vector_id)
            new_index.add(vector.reshape(1, -1))
            
            # РљРѕРїРёСЂСѓРµРј РјРµС‚Р°РґР°РЅРЅС‹Рµ
            new_metadata["vectors"][vector_id] = meta
        
        # РЎРѕС…СЂР°РЅСЏРµРј
        faiss.write_index(new_index, f"{output_dir}/{source}_index.faiss")
        json.dump(new_metadata, open(f"{output_dir}/{source}_metadata.json", "w"))
```
